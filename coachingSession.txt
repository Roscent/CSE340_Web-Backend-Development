const getOrganizationDetails = async (organizationId) => {

      SELECT
        organization_id,
        name,
        description,
        contact_email,
        logo_filename
      FROM organization
      WHERE organization_id = $1;

      return result.rows.length > 0 ? result.rows[0] : null;
